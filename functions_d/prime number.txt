#include <stdio.h>

int is_prime(int num)
{
    int i,count=0;
    for(i=1;i<=num;i++)
    {
      if(num%i==0)
      {
          count++;
      }
    }
  return count;
}

int main()
{
    int num;
    printf("enter the number");
    scanf("%d",&num);
    
  if(is_prime(num) == 2)
  printf("%d is a prime number.", num);
  else
  printf("%d is not a prime number.", num);

  return 0;
    
}
